#PBS -l walltime=%s
#PBS -l nodes=%s
#PBS -N %s
hostname
DIR="%s"
TDIR="%s"
cd $HOME/jscfi/"$DIR"/"$TDIR" || exit 1
date >> log
hostname >> log
echo "PBS_O_WORKDIR=" >> log
cp "$PBS_NODEFILE" pbs-nodes
cp pbs-nodes source.c/
mv input.txt ./


echo "server `date +%%s.%%N` -1 task_started" >> stat
$HOME/jscfi/bin/diststatcollect pbs-nodes >> stat &
STATCOLLECT_PID=$!

/usr/bin/time -f '%%e' -o timing make -C source.c run %s 2> stderr > stdout
mv source.c/output.txt ./
date > completed

kill $STATCOLLECT_PID
echo "server `date +%%s.%%N` -1 task_stopped" >> stat
